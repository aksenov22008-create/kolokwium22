Klasy i obiekty podstawy

Klasa = szablon. Obiekt = konkretna instancja klasy.
public class Samochod {
    // pola (stan obiektu)
    private String marka;
    private int rok;

    // konstruktor - wywoływany przy new Samochod(...)
    public Samochod(String marka, int rok) {
        this.marka = marka;
        this.rok = rok;
    }

    // metoda (zachowanie obiektu)
    public String getInfo() {
        return marka + " (" + rok + ")";
    }
}

// użycie:
Samochod s = new Samochod("Toyota", 2020);
s.getInfo(); // "Toyota (2020)"
Na kolokwium: every public class must be in its own .java file with the same name.

Enkapsulacja i konstruktory podstawy

Ukrywaj pola (private), udostępniaj przez metody (gettery/settery).
public class Osoba {
    private final String imie;   // final = ustawiane raz
    private int wiek;

    // konstruktor z walidacją
    public Osoba(String imie, int wiek) {
        if (wiek < 0) throw new IllegalArgumentException("Wiek < 0");
        this.imie = imie;
        this.wiek = wiek;
    }

    // getter - tylko odczyt
    public String getImie() { return imie; }

    // setter - z możliwością walidacji
    public void setWiek(int wiek) {
        if (wiek < 0) throw new IllegalArgumentException("Wiek < 0");
        this.wiek = wiek;
    }

    // toString - wywołuje się przy System.out.println(obj)
    @Override public String toString() {
        return imie + ", wiek: " + wiek;
    }
}

        super(marka);   // PIERWSZE! wywołuje Pojazd(marka)

        return super.opis() + ", drzwi: " + drzwi; // wywołuje wersję Pojazd

Klasa abstrakcyjna na każdym kolokwium

Nie można jej instancjonować. Definiuje metody bez implementacji.
public abstract class Ksztalt {
    private String kolor;

    public Ksztalt(String kolor) { this.kolor = kolor; }
    public String getKolor() { return kolor; }

    // abstrakcyjna = brak ciała, MUSI być nadpisana w podklasie
    public abstract double pole();
    public abstract String toSvg();

    // zwykła metoda - podklasy dziedziczą gotową implementację
    public void wypisz() {
        System.out.println("Pole: " + pole()); // wywoła właściwą wersję
    }
}

public class Okrag extends Ksztalt {
    private double r;

    public Okrag(String kolor, double r) {
        super(kolor);
        this.r = r;
    }

    @Override public double pole() { return Math.PI * r * r; }
    @Override public String toSvg() {
        return "<circle r='" + r + "'/>";
    }
}

// new Ksztalt() → BŁĄD kompilacji
Ksztalt k = new Okrag("red", 5);  // OK - podklasa konkretna

Wyjątki kluczowe

Checked (kompilator wymaga obsługi) vs Unchecked (brak wymogu).
// CHECKED - extends Exception
class BrakException extends Exception {
    private final String nazwa;
    public BrakException(String nazwa) {
        super(nazwa);         // getMessage() zwróci 'nazwa'
        this.nazwa = nazwa;
    }
    public String getNazwa() { return nazwa; }
}

// UNCHECKED - extends RuntimeException (nie wymaga throws)
class BrakZwyciezcyException extends RuntimeException {
    public BrakZwyciezcyException() { super("Brak zwycięzcy"); }
}

// PROPAGACJA przez throws (checked musi być zadeklarowany)
void metoda() throws BrakException {
    throw new BrakException("coś"); // rzucamy
}

// PRZECHWYCENIE
try {
    metoda();
} catch (BrakException e) {
    e.getMessage(); // "coś"
} catch (IOException | NullPointerException e) { // multicatch
    e.getClass().getSimpleName();
} finally {
    // zawsze się wykonuje (sprzątanie)
}
Na kolokwium: getMessage() musi zwracać szukaną nazwę/wartość - bo task często sprawdza assertThrows + e.getMessage().

Programowanie generyczne ważne

Klasy i metody działające na dowolnych typach - bezpieczne typowo.
// Klasa generyczna
class Para<A, B> {
    private final A pierwszy;
    private final B drugi;
    public Para(A a, B b) { pierwszy = a; drugi = b; }
    public A getPierwszy() { return pierwszy; }
}

Para<String, Integer> p = new Para<>("wynik", 42);

// Metoda generyczna
public static <T extends Comparable<T>> T max(T a, T b) {
    return a.compareTo(b) >= 0 ? a : b;
}
max("alpha", "beta"); // "beta"
max(10, 20);          // 20

// Wildcard - przyjmuje listę dowolnego podtypu Number
double sum(List<? extends Number> lista) {
    return lista.stream().mapToDouble(Number::doubleValue).sum();
}
// akceptuje: List<Integer>, List<Double>, List<Float>

Enum + Record nowe

Enum = skończony zbiór wartości. Record = niezmienialny nośnik danych.
// ENUM z metodami (kolokwium 2024: Mode { HOUR_12, HOUR_24 })
public enum Tryb {
    DWUDZIESTOCZTERY, DWUNASCIE;

    public String getOpis() {  // enum może mieć metody
        return switch (this) {
            case DWUDZIESTOCZTERY -> "24h";
            case DWUNASCIE -> "12h";
        };
    }
}

Tryb t = Tryb.DWUDZIESTOCZTERY;
Tryb.values();  // [DWUDZIESTOCZTERY, DWUNASCIE]
Tryb.valueOf("DWUNASCIE"); // Tryb.DWUNASCIE

// RECORD (Java 16+) - automatycznie: konstruktor, gettery, equals, toString
// (kolokwium 2025: record Candidate(String name){})
record Kandydat(String nazwa, int numer) {
    // gettery: kandydat.nazwa(), kandydat.numer()
    // opcjonalnie: własna logika
    public String nazwisko() { return nazwa.split(" ")[1]; }
}

Kandydat k = new Kandydat("Jan Kowalski", 1);
k.nazwa();   // "Jan Kowalski"
k.numer();   // 1

// record jako klucz w Map (equals i hashCode są automatyczne)
Map<Kandydat, Integer> glosy = new HashMap<>();
glosy.put(k, 5000);

Wzorce z kolokwiów gotowe do użycia

Skopiuj i dostosuj — te struktury powtarzają się co roku.
// ① STATYCZNA METODA FABRYCZNA fromCsv() - pojawia się zawsze
public static MojaKlasa fromCsv(Path sciezka) {
    try {
        List<String> linie = Files.readAllLines(sciezka);
        String nazwa = linie.get(0).trim();    // linia 0 = nazwa
        // linia 1 = nagłówek - pomijamy
        String[] wartosci = linie.get(2).split(";");
        return new MojaKlasa(nazwa, wartosci);
    } catch (IOException e) {
        throw new RuntimeException(e); // checked → unchecked
    }
}

// ② WYSZUKIWANIE PO PREFIKSIE + własny wyjątek (kolokwium 2023)
public static Produkt getProdukt(String prefiks)
        throws NiejednoznacznyWyjatek, IndexOutOfBoundsException {
    List<Produkt> pasujace = lista.stream()
            .filter(p -> p.getNazwa().startsWith(prefiks))
            .collect(Collectors.toList());
    if (pasujace.isEmpty()) throw new IndexOutOfBoundsException(prefiks);
    if (pasujace.size() > 1) {
        List<String> nazwy = pasujace.stream().map(Produkt::getNazwa)
                                      .collect(Collectors.toList());
        throw new NiejednoznacznyWyjatek(nazwy);
    }
    return pasujace.get(0);
}

// ③ ZWYCIĘZCA - >50% lub wyjątek (kolokwium 2025)
public Kandydat winner() {
    int suma = votes.values().stream().mapToInt(Integer::intValue).sum();
    return votes.entrySet().stream()
        .filter(e -> e.getValue() * 100.0 / suma > 50)
        .map(Map.Entry::getKey)
        .findFirst()
        .orElseThrow(BrakZwyciezcyException::new);
}
