Создай обычный Maven-проект. Нам понадобится всего один класс для всей логики и один класс для запуска. Создавать папки (пакеты) будем внутри src/main/java.
Plaintext

Имя_Проекта
 └── src
      └── main
           └── java

                │    └── ImageProcessor.java  <-- Вся логика (Zadanie 1-6)
                └── Main.java                 <-- Запуск и сравнение времени

🛠 Часть 1. Класс обработки изображений (ImageProcessor.java)

Создай папку image внутри src/main/java, а в ней — файл ImageProcessor.java.
Ниже приведен полный код для этого класса, разбитый по заданиям с объяснением каждой строчки.
Java

package image;

import java.awt.Color;
import java.awt.Graphics2D;
import java.awt.image.BufferedImage;
import java.io.File;
import java.io.IOException;
import javax.imageio.ImageIO;
import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;
import java.util.concurrent.TimeUnit;

public class ImageProcessor {
    // Поле класса, в котором хранится картинка в оперативной памяти
    private BufferedImage image;

    // Снаружи нам нужен доступ к картинке (например, чтобы передать в метод построения графика)
    public BufferedImage getImage() {
        return image;
    }

    // ==========================================
    // Zadanie 1. Чтение и сохранение картинки
    // ==========================================

    // Метод читает файл с диска и записывает в поле класса image
    public void loadImage(String path) throws IOException {
        // ImageIO.read сам открывает файл по пути path и превращает его в объект BufferedImage
        this.image = ImageIO.read(new File(path));
    }

    // Метод берет картинку из поля класса image и сохраняет её на диск
    public void saveImage(String path, String format) throws IOException {
        // ImageIO.write принимает: объект картинки, формат (например, "png") и файл куда сохранить
        ImageIO.write(this.image, format, new File(path));
    }

    // ==========================================
    // Zadanie 2. Однопоточное изменение яркости
    // ==========================================

    public void brightenSingleThread(int value) {
        int width = image.getWidth();   // Получаем ширину картинки в пикселях
        int height = image.getHeight(); // Получаем высоту картинки в пикселях

        // Проходим вложенным циклом по каждому пикселю (строка за строкой, столбец за столбцом)
        for (int y = 0; y < height; y++) {
            for (int x = 0; x < width; x++) {
                // Изменяем яркость конкретного пикселя по координатам (x, y)
                changePixelBrightness(x, y, value);
            }
        }
    }

    // Вспомогательный метод для изменения яркости ОДНОГО пикселя
    private void changePixelBrightness(int x, int y, int value) {
        int rgb = image.getRGB(x, y); // Получаем закодированный цвет пикселя (содержит R, G, B)
        Color color = new Color(rgb); // Превращаем число в удобный объект Color

        // Добавляем значение яркости к каждому каналу, ограничивая диапазон от 0 до 255
        int r = Math.min(255, Math.max(0, color.getRed() + value));
        int g = Math.min(255, Math.max(0, color.getGreen() + value));
        int b = Math.min(255, Math.max(0, color.getBlue() + value));

        Color newColor = new Color(r, g, b); // Собираем новый цвет из измененных каналов
        image.setRGB(x, y, newColor.getRGB()); // Записываем новый цвет обратно в картинку
    }

    // ==========================================
    // Zadanie 3. Разделение работы вручную на потоки (Thread)
    // ==========================================

    public void brightenMultiThread(int value) throws InterruptedException {
        // Runtime.getRuntime().availableProcessors() автоматически узнает число ядер твоего процессора
        int cores = Runtime.getRuntime().availableProcessors();
        Thread[] threads = new Thread[cores]; // Создаем массив для потоков

        int height = image.getHeight();
        int width = image.getWidth();
        int rowsPerThread = height / cores; // Вычисляем, сколько строк картинки достанется каждому потоку

        // Делим картинку на горизонтальные полосы по числу ядер процессора
        for (int i = 0; i < cores; i++) {
            final int startY = i * rowsPerThread; // Начальная строка полосы для этого потока
            // Последний поток забирает все оставшиеся строчки (если высота не делится нацело)
            final int endY = (i == cores - 1) ? height : (i + 1) * rowsPerThread;

            // Создаем поток и передаем ему задачу в виде лямбда-выражения () -> { ... }
            threads[i] = new Thread(() -> {
                for (int y = startY; y < endY; y++) {
                    for (int x = 0; x < width; x++) {
                        changePixelBrightness(x, y, value);
                    }
                }
            });
            threads[i].start(); // Запускаем поток параллельно основным
        }

        // Главный поток (Main) должен подождать, пока ВСЕ созданные потоки завершат работу
        for (Thread thread : threads) {
            thread.join(); // Ждем завершения конкретного потока
        }
    }

    // ==========================================
    // Zadanie 4. Использование пулы потоков (ExecutorService)
    // ==========================================

    public void brightenThreadPool(int value) throws InterruptedException {
        int cores = Runtime.getRuntime().availableProcessors();
        // Создаем пул потоков, фиксированный под размер ядер процессора
        ExecutorService executor = Executors.newFixedThreadPool(cores);

        int height = image.getHeight();
        int width = image.getWidth();

        // По условию задания: один поток (одна задача в пуле) обрабатывает ровно ОДИН вiersz (строку)
        for (int y = 0; y < height; y++) {
            final int currentRow = y; // Запоминаем номер текущей строки
            
            // Отправляем задачу на обработку строки в пул потоков
            executor.submit(() -> {
                for (int x = 0; x < width; x++) {
                    changePixelBrightness(x, currentRow, value);
                }
            });
        }

        executor.shutdown(); // Говорим пулу, что новых задач не будет, пора закрываться после выполнения текущих
        executor.awaitTermination(1, TimeUnit.HOURS); // Ждем завершения всех задач (максимум 1 час)
    }

    // ==========================================
    // Zadanie 5. Подсчет гистограммы выбранного канала
    // ==========================================

    // Канал: 0 - Red, 1 - Green, 2 - Blue
    public int[] calculateHistogram(int channel) throws InterruptedException {
        // Гистограмма — это массив из 256 ячеек (по числу возможных уровней цвета от 0 до 255)
        int[] histogram = new int[256];
        int cores = Runtime.getRuntime().availableProcessors();
        ExecutorService executor = Executors.newFixedThreadPool(cores);

        int height = image.getHeight();
        int width = image.getWidth();

        // Тоже делим картинку построчно через пул потоков
        for (int y = 0; y < height; y++) {
            final int currentRow = y;
            executor.submit(() -> {
                for (int x = 0; x < width; x++) {
                    int rgb = image.getRGB(x, currentRow);
                    Color color = new Color(rgb);
                    int colorValue = 0;

                    // Выбираем значение из нужного канала
                    if (channel == 0) colorValue = color.getRed();
                    else if (channel == 1) colorValue = color.getGreen();
                    else if (channel == 2) colorValue = color.getBlue();

                    // Поскольку потоки работают одновременно, синхронизируем доступ к общему массиву,
                    // чтобы они не перезаписали данные друг друга в один момент времени.
                    synchronized (histogram) {
                        histogram[colorValue]++; // Увеличиваем счетчик для конкретного оттенка
                    }
                }
            });
        }

        executor.shutdown();
        executor.awaitTermination(1, TimeUnit.HOURS);
        return histogram; // Возвращаем заполненный массив гистограммы
    }

    // ==========================================
    // Zadanie 6. Генерация изображения-графика гистограммы
    // ==========================================

    public static BufferedImage generateHistogramChart(int[] histogram) {
        int width = 256; // Ширина графика равна количеству оттенков цвета
        int height = 150; // Высота графика в пикселях
        // Создаем новую пустую картинку для графика (тип INT_RGB)
        BufferedImage chart = new BufferedImage(width, height, BufferedImage.TYPE_INT_RGB);
        
        // Graphics2D — это инструмент («кисть»), позволяющий рисовать геометрические фигуры на картинке
        Graphics2D g = chart.createGraphics();

        // Заливаем весь фон белым цветом
        g.setColor(Color.WHITE);
        g.fillRect(0, 0, width, height);

        // Ищем максимальное значение в гистограмме, чтобы график поместился по высоте и не вылез за края
        int max = 0;
        for (int val : histogram) {
            if (val > max) max = val;
        }

        // Рисуем график черным цветом в виде вертикальных линий (столбцов)
        g.setColor(Color.BLACK);
        for (int x = 0; x < width; x++) {
            // Вычисляем высоту столбца пропорционально максимальному значению
            int barHeight = (int) (((double) histogram[x] / max) * (height - 10));
            // Рисуем линию снизу вверх: от (X, общая_высота) до (X, общая_высота - высота_столбца)
            g.drawLine(x, height, x, height - barHeight);
        }

        g.dispose(); // Освобождаем ресурсы кисти рисунка
        return chart; // Возвращаем готовую картинку с графиком
    }
}

🚀 Часть 2. Класс запуска и сравнения времени (Main.java)

Создай файл Main.java прямо внутри папки src/main/java. Он запускает методы изменения яркости, замеряет время работы в наносекундах (System.nanoTime()) и строит график.

    💡 Перед запуском: Положи любую картинку в корень проекта под именем input.png.

Java

import image.ImageProcessor;
import java.awt.image.BufferedImage;
import java.io.File;
import javax.imageio.ImageIO;

public class Main {
    public static void main(String[] args) {
        try {
            // Создаем объект процессора изображений
            ImageProcessor processor = new ImageProcessor();

            // 1. Тест однопоточного метода
            processor.loadImage("input.png"); // Загружаем картинку
            long start1 = System.nanoTime();   // Засекаем время СТАРТА
            processor.brightenSingleThread(50); // Увеличиваем яркость на 50 пунктов
            long end1 = System.nanoTime();     // Засекаем время ОКОНЧАНИЯ
            System.out.println("Однопоточный метод: " + (end1 - start1) / 1_000_000.0 + " мс");
            processor.saveImage("output_single.png", "png"); // Сохраняем результат

            // 2. Тест многопоточного метода (разделение по ядрам)
            processor.loadImage("input.png");
            long start2 = System.nanoTime();
            processor.brightenMultiThread(50);
            long end2 = System.nanoTime();
            System.out.println("Многопоточный (Thread): " + (end2 - start2) / 1_000_000.0 + " мс");

            // 3. Тест метода с пулом потоков (построчно)
            processor.loadImage("input.png");
            long start3 = System.nanoTime();
            processor.brightenThreadPool(50);
            long end3 = System.nanoTime();
            System.out.println("Пул потоков (ThreadPool): " + (end3 - start3) / 1_000_000.0 + " мс");

            // 4. Тест генерации гистограммы и графика (Zadanie 5 и 6)
            // Считаем гистограмму для Красного канала (0)
            int[] redHistogram = processor.calculateHistogram(0);
            // Генерируем изображение графика на основе полученного массива данных
            BufferedImage chartImage = ImageProcessor.generateHistogramChart(redHistogram);
            // Сохраняем картинку-график на диск
            ImageIO.write(chartImage, "png", new File("histogram_chart.png"));
            System.out.println("Гистограмма успешно сохранена в файл: histogram_chart.png");

        } catch (Exception e) {
            // Если файла input.png не окажется на месте, программа выведет ошибку:
            System.out.println("Произошла ошибка: " + e.getMessage());
            e.printStackTrace();
        }
    }
}
